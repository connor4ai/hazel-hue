    <script>
        // Elements
        const uploadZone = document.getElementById('uploadZone');
        const fileInput = document.getElementById('fileInput');
        const photoGrid = document.getElementById('photoGrid');
        const photoSlots = document.querySelectorAll('.photo-slot');
        const analyzeBtn = document.getElementById('analyzeBtn');
        const loadingOverlay = document.getElementById('loadingOverlay');

        let uploadedFiles = [null, null, null];

        // Upload zone click
        uploadZone.addEventListener('click', () => {
            if (!uploadZone.classList.contains('active')) {
                fileInput.click();
            }
        });

        // File input change
        fileInput.addEventListener('change', (e) => {
            handleFiles(e.target.files);
        });

        // Drag and drop
        uploadZone.addEventListener('dragover', (e) => {
            e.preventDefault();
            uploadZone.classList.add('dragover');
        });

        uploadZone.addEventListener('dragleave', () => {
            uploadZone.classList.remove('dragover');
        });

        uploadZone.addEventListener('drop', (e) => {
            e.preventDefault();
            uploadZone.classList.remove('dragover');
            handleFiles(e.dataTransfer.files);
        });

        // Handle files
        function handleFiles(files) {
            const imageFiles = Array.from(files).filter(file => file.type.startsWith('image/'));
            
            let index = uploadedFiles.findIndex(f => f === null);
            
            imageFiles.forEach(file => {
                if (index < 3 && index !== -1) {
                    uploadedFiles[index] = file;
                    displayImage(file, index);
                    index = uploadedFiles.findIndex(f => f === null);
                }
            });

            updateUI();
        }

        // Display image
        function displayImage(file, index) {
            const reader = new FileReader();
            reader.onload = (e) => {
                const slot = photoSlots[index];
                slot.innerHTML = `
                    <img src="${e.target.result}" alt="Photo ${index + 1}">
                    <button class="remove" onclick="removePhoto(${index})">×</button>
                    <div class="photo-number">${index + 1}</div>
                `;
                slot.classList.add('filled');
            };
            reader.readAsDataURL(file);
        }

        // Remove photo
        function removePhoto(index) {
            uploadedFiles[index] = null;
            const slot = photoSlots[index];
            slot.classList.remove('filled');
            slot.innerHTML = `<div class="photo-number">${index + 1}</div>`;
            updateUI();
        }

        // Update UI
        function updateUI() {
            const uploadCount = uploadedFiles.filter(f => f !== null).length;
            
            // Show/hide elements
            if (uploadCount > 0) {
                photoGrid.classList.add('visible');
                analyzeBtn.classList.add('visible');
                uploadZone.classList.add('active');
            } else {
                photoGrid.classList.remove('visible');
                analyzeBtn.classList.remove('visible');
                uploadZone.classList.remove('active');
            }

            // Enable/disable analyze button
            if (uploadCount === 3) {
                analyzeBtn.classList.add('ready');
            } else {
                analyzeBtn.classList.remove('ready');
            }

            // Update upload zone text
            if (uploadCount > 0 && uploadCount < 3) {
                uploadZone.querySelector('.upload-text').textContent = `Add ${3 - uploadCount} more photo${3 - uploadCount > 1 ? 's' : ''}`;
                uploadZone.querySelector('.upload-subtext').textContent = 'Click or drop to continue';
            } else if (uploadCount === 3) {
                uploadZone.querySelector('.upload-text').textContent = 'All photos uploaded';
                uploadZone.querySelector('.upload-subtext').textContent = 'Ready for analysis';
            }
        }

        // Analyze button click
        analyzeBtn.addEventListener('click', () => {
            if (analyzeBtn.classList.contains('ready')) {
                loadingOverlay.classList.add('active');
                
                // Simulate processing
                setTimeout(() => {
                    // In real app, proceed to results
                    alert('Analysis complete! Proceeding to results...');
                    loadingOverlay.classList.remove('active');
                }, 2000);
            }
        });

        // Window load animation
        window.addEventListener('load', () => {
            document.body.style.opacity = '1';
        });
    </script>
