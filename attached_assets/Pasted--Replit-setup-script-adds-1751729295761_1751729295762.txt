# 2️⃣  lab_utils.py  – helper that returns {skin_LAB, hair_LAB, eye_LAB}
cat > lab_utils.py <<'PY'
import cv2, mediapipe as mp, numpy as np
from skimage import color

# Landmark index groups (MediaPipe FaceMesh)
_SKIN = [33,133,362,263,1,199,97,326]               # cheeks + forehead
_HAIR = [10,338,297,332,284,251,9,8]                # hairline polygon
_IRIS = [468,469,470,471,472,473,474,475,476,477]   # both irises

def _mask(lms, idxs, h, w):
    pts = np.array([(int(lms.landmark[i].x*w),
                     int(lms.landmark[i].y*h)) for i in idxs])
    mask = np.zeros((h, w), np.uint8)
    cv2.fillConvexPoly(mask, pts, 255)
    return mask.astype(bool)

def _mean_lab(region_rgb):
    lab = color.rgb2lab(region_rgb.reshape(-1,3) / 255.0)
    return [round(float(x), 2) for x in lab.mean(axis=0)]

def extract_lab(img_path:str) -> dict:
    """Return {'skin_LAB': [...], 'hair_LAB': [...], 'eye_LAB': [...]}."""
    bgr = cv2.imread(img_path)
    if bgr is None:
        raise FileNotFoundError(img_path)
    rgb = cv2.cvtColor(bgr, cv2.COLOR_BGR2RGB)
    h, w = rgb.shape[:2]

    with mp.solutions.face_mesh.FaceMesh(static_image_mode=True) as fm:
        res = fm.process(rgb)
        if not res.multi_face_landmarks:
            raise ValueError("No face detected in " + img_path)
        lms = res.multi_face_landmarks[0]

    return {
        "skin_LAB": _mean_lab(rgb[_mask(lms,_SKIN,h,w)]),
        "hair_LAB": _mean_lab(rgb[_mask(lms,_HAIR,h,w)]),
        "eye_LAB" : _mean_lab(rgb[_mask(lms,_IRIS,h,w)]),
    }
PY
echo "✅  lab_utils.py written."

# 3️⃣  main.py – quick CLI test
cat > main.py <<'PY'
import sys, json
from lab_utils import extract_lab

files = sys.argv[1:] or ["photo_A.jpg", "photo_B.jpg", "photo_C.jpg"]
out = {f: extract_lab(f) for f in files}
print(json.dumps(out, indent=2))
PY
