                navigator.clipboard.writeText(colorCode).then(() => {

                    feedback.textContent = 'Copied!';

                        backdrop-filter: blur(8px);
                        border: 1px solid rgba(255, 255, 255, 0.3);
                    `;
                    document.body.appendChild(feedback);
