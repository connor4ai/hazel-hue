"""
lab_utils.py  – Extract average CIE L*a*b* values for skin, hair, and eyes
from a single head-and-shoulders photo.  Designed for Replit (no GUI libs).
"""
import cv2, mediapipe as mp, numpy as np
from skimage import color

# FaceMesh landmark clusters
_SKIN = [33,133,362,263,1,199,97,326]                # cheeks + forehead
_HAIR = [10,338,297,332,284,251,9,8]                 # rough hairline
_IRIS = [468,469,470,471,472,473,474,475,476,477]    # both irises

def _mask(lms, idxs, h, w):
    pts = np.array([(int(lms.landmark[i].x*w),
                     int(lms.landmark[i].y*h)) for i in idxs])
    mask = np.zeros((h, w), np.uint8)
    cv2.fillConvexPoly(mask, pts, 255)
    return mask.astype(bool)

def _mean_lab(region_rgb):
    lab = color.rgb2lab(region_rgb.reshape(-1,3) / 255.0)
    return [round(float(x),2) for x in lab.mean(axis=0)]

def extract_lab(path:str) -> dict:
    """Return {'skin_LAB':[L,a,b], 'hair_LAB':[...], 'eye_LAB':[...]}"""
    bgr = cv2.imread(path); assert bgr is not None, f"Cannot open {path}"
    rgb = cv2.cvtColor(bgr, cv2.COLOR_BGR2RGB); h,w = rgb.shape[:2]

    with mp.solutions.face_mesh.FaceMesh(static_image_mode=True) as fm:
        res = fm.process(rgb)
        if not res.multi_face_landmarks:
            raise ValueError("No face detected in " + path)
        lms = res.multi_face_landmarks[0]

    return {
        "skin_LAB": _mean_lab(rgb[_mask(lms,_SKIN,h,w)]),
        "hair_LAB": _mean_lab(rgb[_mask(lms,_HAIR,h,w)]),
        "eye_LAB" : _mean_lab(rgb[_mask(lms,_IRIS,h,w)]),
    }
PY

import os, json, numpy as np, openai, tempfile
from pathlib import Path
from lab_utils import extract_lab

openai.api_key = os.getenv("OPENAI_API_KEY")

def analyse_photos(file_paths:list[str]) -> dict:
    """file_paths = 3 local image paths"""
    labs = [extract_lab(p) for p in file_paths]

    # average each key across the three photos
    def mean_key(k): return [round(float(x),2) for x in np.mean(
        np.array([d[k] for d in labs]), axis=0)]
    avg_lab = {k: mean_key(k) for k in labs[0]}

    prompt = (
        "You are a certified personal-color analyst (12-season system).\\n"
        f"Here are averaged CIE Lab readings: {json.dumps(avg_lab)}\\n"
        'Return JSON like {"season":"Bright Spring","confidence":95}.'
    )
    res = openai.chat.completions.create(
        model="gpt-4o-mini", temperature=0,
        messages=[{"role":"user","content":prompt}]
    )
    return json.loads(res.choices[0].message.content)

# Example Flask route
@app.post("/upload")
def upload():
    photos = [request.files[f] for f in ("photo1","photo2","photo3")]
    with tempfile.TemporaryDirectory() as tmp:
        paths = []
        for p in photos:
            dst = Path(tmp)/p.filename; p.save(dst); paths.append(str(dst))
        result = analyse_photos(paths)
    return jsonify(result)
