    <script>
        // Add click to copy functionality
        document.querySelectorAll('.color-swatch').forEach(swatch => {
            swatch.addEventListener('click', function() {
                const colorCode = this.querySelector('.color-code').textContent;
                navigator.clipboard.writeText(colorCode).then(() => {
                    // Create temporary feedback with dark autumn styling
                    const feedback = document.createElement('div');
                    feedback.textContent = 'Copied!';
                    feedback.style.cssText = `
                        position: fixed;
                        top: 50%;
                        left: 50%;
                        transform: translate(-50%, -50%);
                        background: linear-gradient(135deg, #654321 0%, #3C2415 100%);
                        color: #ffffff;
                        padding: 12px 24px;
                        border-radius: 8px;
                        z-index: 1000;
                        font-weight: 500;
                        font-size: 0.9em;
                        box-shadow: 0 8px 24px rgba(0, 0, 0, 0.3);
                        backdrop-filter: blur(8px);
                        border: 1px solid rgba(255, 255, 255, 0.3);
                    `;
                    document.body.appendChild(feedback);
                    setTimeout(() => document.body.removeChild(feedback), 1000);
                });
            });
        });
